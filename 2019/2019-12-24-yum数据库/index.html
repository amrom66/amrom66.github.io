作者： linjinbao66 时间： December 24, 2019 分类：

yum 重建数据库解决报错

[root@www~]# cd /var/lib/rpm      # rpmdb所在目录
[root@www rpm]# rm -f __db.*     # 清除原rpmdb文件
[root@www rpm]# rpm --rebuilddb     # 重建rpm数据库
[root@www rpm]# yum clean all     # 清除所有yum的缓存
